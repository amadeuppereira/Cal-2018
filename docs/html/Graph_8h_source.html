888  double R = 6371;
889 
890  double lat1r = v1->getLatitude() * M_PI / 180.0;
891  double lat2r = v2->getLatitude() * M_PI / 180.0;
892  double lon1r = v1->getLongitude() * M_PI / 180.0;
893  double lon2r = v2->getLongitude() * M_PI / 180.0;
894 
895  double latdiff = abs(lat2r - lat1r);
